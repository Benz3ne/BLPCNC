local inst = mc.mcGetInstance()
local currentMode = mc.mcCntlGetPoundVar(inst, 402)
if currentMode == 0 then
    mc.mcCntlSetPoundVar(inst, 402, 1)
    mc.mcCntlSetPoundVar(inst, 410, 0)  -- Clear any override
    scr.SetProperty("btnDustBootAuto", "Bg Color", "#00FF00")
    mc.mcCntlSetLastError(inst, "Dust boot AUTO mode")
else
    mc.mcCntlSetPoundVar(inst, 402, 0)
    scr.SetProperty("btnDustBootAuto", "Bg Color", "#FF0000")
    mc.mcCntlSetLastError(inst, "Dust boot MANUAL mode")
end
